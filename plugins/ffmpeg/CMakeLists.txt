# ffmpeg plugin

find_package(FFmpeg REQUIRED)

file(GLOB_RECURSE KHOPPER_PLUGIN_FFMPEG_HEADERS RELATIVE ${CMAKE_CURRENT_SOURCE_DIR} *.hpp)
file(GLOB_RECURSE KHOPPER_PLUGIN_FFMPEG_SOURCES RELATIVE ${CMAKE_CURRENT_SOURCE_DIR} *.cpp)
file(GLOB_RECURSE KHOPPER_PLUGIN_FFMPEG_FORMS RELATIVE ${CMAKE_CURRENT_SOURCE_DIR} *.ui)
if(NOT WIN32)
    list(REMOVE_ITEM KHOPPER_CORE_SOURCES wfile.hpp wfile.cpp)
endif()

include_directories(${CMAKE_CURRENT_BINARY_DIR} ${FFMPEG_INCLUDE_DIR})

qt4_wrap_ui(KHOPPER_PLUGIN_FFMPEG_UIC_HEADERS ${KHOPPER_PLUGIN_FFMPEG_FORMS})

source_group("Generated Files" FILES ${KHOPPER_PLUGIN_FFMPEG_UIC_HEADERS})
source_group("Form Files" REGULAR_EXPRESSION .*\\.ui)

add_library(khopper_plugin_ffmpeg SHARED ${KHOPPER_PLUGIN_FFMPEG_HEADERS} ${KHOPPER_PLUGIN_FFMPEG_SOURCES} ${KHOPPER_PLUGIN_FFMPEG_UIC_HEADERS})
target_link_libraries(khopper_plugin_ffmpeg ${FFMPEG_LIBRARIES})
khopper_add_plugin(khopper_plugin_ffmpeg)

install(TARGETS khopper_plugin_ffmpeg
    RUNTIME DESTINATION "plugins"
    LIBRARY DESTINATION "plugins"
    ARCHIVE DESTINATION "lib")
