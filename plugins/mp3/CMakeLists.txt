# mp3 panel

find_package(Lame REQUIRED)
find_package(TagLib REQUIRED)

file(GLOB_RECURSE KHOPPER_PLUGIN_PANEL_MP3_HEADERS RELATIVE ${CMAKE_CURRENT_SOURCE_DIR} *.hpp)
file(GLOB_RECURSE KHOPPER_PLUGIN_PANEL_MP3_SOURCES RELATIVE ${CMAKE_CURRENT_SOURCE_DIR} *.cpp)
file(GLOB_RECURSE KHOPPER_PLUGIN_PANEL_MP3_FORMS RELATIVE ${CMAKE_CURRENT_SOURCE_DIR} *.ui)

include_directories(
    ${LAME_INCLUDE_DIR}
    ${TAGLIB_INCLUDE_DIR}
    ${CMAKE_CURRENT_BINARY_DIR}
)

qt4_wrap_ui(KHOPPER_PLUGIN_PANEL_MP3_UIC_HEADERS ${KHOPPER_PLUGIN_PANEL_MP3_FORMS})

source_group("Generated Files" FILES ${KHOPPER_PLUGIN_PANEL_MP3_UIC_HEADERS})
source_group("Form Files" REGULAR_EXPRESSION .*\\.ui)

add_library(khopper_panel_mp3 SHARED ${KHOPPER_PLUGIN_PANEL_MP3_HEADERS} ${KHOPPER_PLUGIN_PANEL_MP3_SOURCES} ${KHOPPER_PLUGIN_PANEL_MP3_UIC_HEADERS})
khopper_add_plugin(khopper_panel_mp3)
target_link_libraries(khopper_panel_mp3 ${LAME_LIBRARIES} ${TAGLIB_LIBRARIES})

install(TARGETS khopper_panel_mp3 DESTINATION
    RUNTIME DESTINATION "plugins"
    LIBRARY DESTINATION "plugins"
    ARCHIVE DESTINATION "lib")
