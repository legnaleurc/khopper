# ogg panel

find_package(OggVorbis REQUIRED)

file(GLOB_RECURSE KHOPPER_PLUGIN_PANEL_OGG_HEADERS RELATIVE ${CMAKE_CURRENT_SOURCE_DIR} *.hpp)
file(GLOB_RECURSE KHOPPER_PLUGIN_PANEL_OGG_SOURCES RELATIVE ${CMAKE_CURRENT_SOURCE_DIR} *.cpp)
file(GLOB_RECURSE KHOPPER_PLUGIN_PANEL_OGG_FORMS RELATIVE ${CMAKE_CURRENT_SOURCE_DIR} *.ui)

include_directories(
    "${KHOPPER_XIPH_DIR}/l_flac"
    ${OGGVORBIS_INCLUDE_DIR}
    ${CMAKE_CURRENT_BINARY_DIR}
)
add_definitions(-DKHOPPER_XIPH_LIB=${KHOPPER_XIPH_LIB})

qt4_wrap_ui(KHOPPER_PLUGIN_PANEL_OGG_UIC_HEADERS ${KHOPPER_PLUGIN_PANEL_OGG_FORMS})

source_group("Generated Files" FILES ${KHOPPER_PLUGIN_PANEL_OGG_UIC_HEADERS})
source_group("Form Files" REGULAR_EXPRESSION .*\\.ui)

add_library(khopper_panel_ogg SHARED ${KHOPPER_PLUGIN_PANEL_OGG_HEADERS} ${KHOPPER_PLUGIN_PANEL_OGG_SOURCES} ${KHOPPER_PLUGIN_PANEL_OGG_UIC_HEADERS})
khopper_add_plugin(khopper_panel_ogg)
target_link_libraries(khopper_panel_ogg ${OGGVORBIS_LIBRARIES})
