# flac panel

file(GLOB_RECURSE KHOPPER_PLUGIN_PANEL_FLAC_HEADERS RELATIVE ${CMAKE_CURRENT_SOURCE_DIR} *.hpp)
file(GLOB_RECURSE KHOPPER_PLUGIN_PANEL_FLAC_SOURCES RELATIVE ${CMAKE_CURRENT_SOURCE_DIR} *.cpp)
file(GLOB_RECURSE KHOPPER_PLUGIN_PANEL_FLAC_FORMS RELATIVE ${CMAKE_CURRENT_SOURCE_DIR} *.ui)

include_directories(
    "${KHOPPER_XIPH_DIR}/l_flac"
    ${CMAKE_CURRENT_BINARY_DIR}
)
add_definitions(-DKHOPPER_XIPH_LIB=${KHOPPER_XIPH_LIB})

qt4_wrap_ui(KHOPPER_PLUGIN_PANEL_FLAC_UIC_HEADERS ${KHOPPER_PLUGIN_PANEL_FLAC_FORMS})

source_group("Generated Files" FILES ${KHOPPER_PLUGIN_PANEL_FLAC_UIC_HEADERS})
source_group("Form Files" REGULAR_EXPRESSION .*\\.ui)

add_library(khopper_panel_flac SHARED ${KHOPPER_PLUGIN_PANEL_FLAC_HEADERS} ${KHOPPER_PLUGIN_PANEL_FLAC_SOURCES} ${KHOPPER_PLUGIN_PANEL_FLAC_MOC_SOURCES} ${KHOPPER_PLUGIN_PANEL_FLAC_UIC_HEADERS})
khopper_add_plugin(khopper_panel_flac)
