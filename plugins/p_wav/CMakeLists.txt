# wav panel

file(GLOB_RECURSE KHOPPER_PLUGIN_PANEL_WAV_HEADERS RELATIVE ${CMAKE_CURRENT_SOURCE_DIR} *.hpp)
file(GLOB_RECURSE KHOPPER_PLUGIN_PANEL_WAV_SOURCES RELATIVE ${CMAKE_CURRENT_SOURCE_DIR} *.cpp)
file(GLOB_RECURSE KHOPPER_PLUGIN_PANEL_WAV_FORMS RELATIVE ${CMAKE_CURRENT_SOURCE_DIR} *.ui)

include_directories(${CMAKE_CURRENT_BINARY_DIR})

qt4_wrap_ui(KHOPPER_PLUGIN_PANEL_WAV_UIC_HEADERS ${KHOPPER_PLUGIN_PANEL_WAV_FORMS})

source_group("Generated Files" FILES ${KHOPPER_PLUGIN_PANEL_WAV_UIC_HEADERS})
source_group("Form Files" REGULAR_EXPRESSION .*\\.ui)

add_library(khopper_panel_wav SHARED ${KHOPPER_PLUGIN_PANEL_WAV_HEADERS} ${KHOPPER_PLUGIN_PANEL_WAV_SOURCES} ${KHOPPER_PLUGIN_PANEL_WAV_UIC_HEADERS})
khopper_add_plugin(khopper_panel_wav)
